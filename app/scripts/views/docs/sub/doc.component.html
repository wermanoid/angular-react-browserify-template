@Component

This decorator defines component. 

    import Component from 'core/component';

    @Component({name: 'navigation'})
    export class NavigationController {
        constructor($mdSidenav) {
            'ngInject';
            this.$mdSidenav = $mdSidenav;
        }

        toggleNav() {
            this.$mdSidenav('left').toggle();
        }
    }
