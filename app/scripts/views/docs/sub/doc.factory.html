@Factory

This decorator defines factory. 

    import Factory from 'core/factory';

    @Factory({name: 'tFactory'})
    export class TFactory {
        constructor($timeout, $log, testApi) {
            'ngInject';
            $timeout(() => $log.info('Timeout from factory', testApi), 0);
            return {bla: 1234};
        }
    }
