@Service

This decorator defines service. 

    import Service from 'core/service';

    @Service({name: 'tService'})
    export class TService {
        constructor($timeout, $log, testApi) {
            'ngInject';
            $timeout(() => $log.info('Timeout from service', testApi), 0);
        }

        show() {
            return 12345;
        }
    }
